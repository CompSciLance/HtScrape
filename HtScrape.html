//production rules
Program ::= Statement <EOF>
Statement ::= ( IntDeclaration | NumberDeclaration | StringDeclaration | lookahead | PrintStatement | PrintFullTableStatement | PrintDataStatement | GetDataStatement | IfStatement | WhileStatement )*
variable ::= <IDENTIFIER>
newVariable ::= <IDENTIFIER>
IntDeclaration ::= <INT> newVariable <EQUALS> integerConstant <SEMICOLON>
NumberDeclaration ::= <NUM> newVariable <EQUALS> numberConstant <SEMICOLON>
ArithmeticStatement ::= variable ( <PLUS_EQUAL> | <MINUS_EQUAL> | <MULTIPLY_EQUAL> | <DIVIDE_EQUAL> ) Operator <SEMICOLON>
lookahead ::= ( ArithmeticStatement | lookahead2 )
lookahead2 ::= ( SimpleAssignment | ComplexAssignment )
SimpleAssignment ::= variable <EQUALS> ( integerConstant | stringConstant ) <SEMICOLON>
ComplexAssignment ::= variable <EQUALS> Expression <SEMICOLON>
StringDeclaration ::= <STRING> newVariable <EQUALS> stringConstant <SEMICOLON>
MultipleParams ::= ( variable | integerConstant | numberConstant | stringConstant ) ( <COMMA> ( variable | integerConstant | numberConstant | stringConstant ) )*
PrintStatement ::= <PRINT> <LEFT_PAREN> variable <RIGHT_PAREN> <SEMICOLON>
PrintFullTableStatement ::= <PRINT_FULL_TABLE> <LEFT_PAREN> MultipleParams <RIGHT_PAREN> <SEMICOLON>
PrintDataStatement ::= <PRINT_DATA> <LEFT_PAREN> MultipleParams <RIGHT_PAREN> <SEMICOLON>
GetDataStatement ::= <GET_DATA> <LEFT_PAREN> MultipleParams <RIGHT_PAREN> <SEMICOLON>
IfStatement ::= <IF> <LEFT_PAREN> Expression <RIGHT_PAREN> <LEFT_CURLY> Statement <RIGHT_CURLY> ( ElseStatement )?
ElseStatement ::= <ELSE> <LEFT_CURLY> Statement <RIGHT_CURLY>
WhileStatement ::= <WHILE> <LEFT_PAREN> Expression <RIGHT_PAREN> <LEFT_CURLY> Statement <RIGHT_CURLY>
Expression ::= Operator
Operator ::= Operand ( ( ( expr_operator Operand ) | ( arith_operator Operand )+ ) )?
Operand ::= ( variable | integerConstant | numberConstant | stringConstant )
expr_operator ::= ( <GREATER_THAN> | <LESS_THAN> | <LESS_EQUAL> | <GREATER_EQUAL> | <DOUBLE_EQUAL> | <NOT_EQUAL> )
arith_operator ::= ( <PLUS> | <MINUS> | <MULTIPLY> | <DIVIDE> )
integerConstant ::= <INTEGER>
numberConstant ::= <NUMBER>
stringConstant ::= <STRING_LITERAL>
